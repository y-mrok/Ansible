#
# 既存パッケージを更新
#
dnf -y install open-vm-tools
dnf -y install epel-release
dnf -y update
reboot

#
# OS を設定
#
dnf -y upgrade
dnf -y install setroubleshoot-server curl lsof wget
sed -i  '/swap/d' /etc/fstab
swapoff -a
firewall-cmd --permanent --zone=public --add-service=https
firewall-cmd --zone=public --add-masquerade --permanent
firewall-cmd --reload
reboot

#
# K3S をセットアップ
#
curl -sfL https://get.k3s.io | sh
cat /etc/systemd/system/k3s.service
systemctl status k3s
kubectl get nodes
kubectl get pods --all-namespaces

#
# AWX operator をデプロイ
#
kubectl config set-context --current --namespace=default
kubectl apply -f https://raw.githubusercontent.com/ansible/awx-operator/0.10.0/deploy/awx-operator.yaml
kubectl get events -w --all-namespaces
kubectl get pods
kubectl logs -f deployment.apps/awx-operator

#
# AWX をデプロイ
#
kubectl create namespace awx
kubectl config set-context --current --namespace=awx
vi myawx.yml

----- ここから myawx.yml の内容 -----
apiVersion: awx.ansible.com/v1beta1
kind: AWX
metadata:
  name: awx
spec:
  ingress_type: Ingress
  route_tls_termination_mechanism: edge
  hostname: awx.example.jp
----- ここまで myawx.yml の内容 -----

cat myawx.yml
kubectl apply -f myawx.yml
kubectl get events -w --all-namespaces
kubectl logs -f deployment.apps/awx-operator -n default
kubectl get pods
kubectl get svc

#
# admin アカウントのパスワードを生成
#
kubectl get secret awx-admin-password -o jsonpath='{.data.password}' | base64 --decode
